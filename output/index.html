from multiprocessing import Process

    def run(self):

worker.start()

PriorityQueue

class PriorityQueue(object):
    def __init__(self):
        self._q = []
        self._count = 0
        self._cv = threading.Condition()

    def __str__(self):
        return str(self._q)

    def __repr__(self):
        return self._q

    def put(self, item, priority):
        with self._cv:
            heapq.heappush(self._q, (-priority,self._count,item))
            self._count += 1
            self._cv.notify()

    def pop(self):
        with self._cv:
            while len(self._q) == 0:
                print("wait...")
                self._cv.wait()
            ret = heapq.heappop(self._q)[-1]
        return ret

线程池进程池

from concurrent.futures import ThreadPoolExecutor,ProcessPoolExecutor

with ThreadPoolExecutor(3) as e:
    # 单独提交任务
    fut = e.submit(fib, 30)
    res = fut.result()
    print(res)
    # 或使用map函数
    with futures.ProcessPoolExecutor(2) as e:
       res = e.map(fib, [35]*2)
       for _ in res:
           print(_)
